[docs]def notTruthTable() -> None:
    """Not truth table.

    Prints a truth table for the not operator.

    Returns:
         None. Only prints out a table.
    """
    print(" _________________________\n",
          "|not A    | Evaluates to:|\n",
          "|_________|______________|\n",
          "|not False| True         |\n",
          "|not True | False        |\n",
          "|_________|______________|\n")
    return None
